--01. Oracle ALTER TABLE ADD column examples
--To add a new column to a table, you use the following syntax:
ALTER TABLE table_name
ADD column_name type constraint;


--Let's create a table named members for the demonstration
CREATE TABLE members(
member_id NUMBER GENERATED BY DEFAULT AS IDENTITY,
first_name VARCHAR2 (50),
last_name VARCHAR2 (50),
PRIMARY KEY (member_id)
);
select * from members;

--01. Oracle ALTER TABLE MODIFY column examples
--To modify the attributes of a column, you use the following syntax:

ALTER TABLE members

How to create Sequence in Oracle 
-------------------------------------
CREATE SEQUENCE seq_employee_id
START WITH 1
INCREMENT BY 1
MAXVALUE 1000
NOCYCLE
NOCACHE;


SELECT seq_employee_id.NEXTVAL FROM DUAL;


create table emp
(
emp_id number default seq_employee_id.NEXTVAL not null,
emp_name varchar2(50),
salary number(18,2)
);

insert into emp (emp_name, salary) values ('Rifath', 50000);
insert into emp (emp_name, salary) values('Rishad', 524145);
insert into emp (emp_name, salary) values ('Rifkath', 542452);

select * from emp 


-------Creating View-------
-------------------
CREATE VIEW ভিউ_নাম AS
SELECT কলাম_নাম1, কলাম_নাম2, ...
FROM টেবিল_নাম
WHERE শর্ত;

CREATE VIEW view_department_10 AS
SELECT EMPLOYEE_ID, FIRST_NAME, LAST_NAME, SALARY
FROM EMPLOYEES
WHERE DEPARTMENT_ID = 10;

SELECT * FROM view_department_10;
----Complex View----
CREATE VIEW view_department_salary AS
SELECT DEPARTMENT_ID, SUM(SALARY) AS TOTAL_SALARY
FROM EMPLOYEES
GROUP BY DEPARTMENT_ID;

SELECT * FROM view_department_salary;

select * from EMPLOYEES

---View আপডেট করা---
কিছু ভিউ আপডেট করা যায় যদি তা updatable view হয়। উদাহরণস্বরূপ:
-----------------------------------------------

UPDATE view_department_10
SET SALARY = SALARY + 1000
WHERE EMPLOYEE_ID = 101;

---View মুছে ফেলা---
ভিউ ডিলিট করতে DROP VIEW কমান্ড ব্যবহার করা হয়:
-------------------------------------------

DROP VIEW view_department_10;
